    "for frame in sample_waveform[start_index_waveform:]:\n",

    "        stop_index_waveform = cont\n",
    "        break\n",
    "    cont += 1\n",
    "\n",
    "original_data_zoom = original_data_value[\n",

    "    start_index_original_data:stop_index_original_data\n",
