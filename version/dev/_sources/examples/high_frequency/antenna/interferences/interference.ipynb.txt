    "        all_colors.append(col)\n",

    "    fig = go.Figure(\n",
    "        data=[\n",

    "                    ],\n",
    "                    line_color=\"darkslategray\",\n",
    "                    fill_color=\"grey\",\n",
    "                    align=[\"left\", \"center\"],\n",
    "                    font=dict(color=\"white\", size=16),\n",
    "                ),\n",
    "                cells=dict(\n",
