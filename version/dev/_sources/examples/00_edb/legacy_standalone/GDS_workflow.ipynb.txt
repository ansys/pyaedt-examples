    "## Case 1: Import a GDS file.\n",
    "\n",

    "\n",
    "- XML_Automation.xml\n",
    "  defines physical information such\n",

    "temp_dir = tempfile.TemporaryDirectory(suffix=\".ansys\")\n",
    "control_fn = \"Model.xml\"\n",
    "gds_fn = \"Model.gds\"\n",
    "layer_map = \"Model.map\"\n",
    "\n",
    "local_path = download_file(\"gds\", destination=temp_dir.name)\n",
    "my_control_file = os.path.join(local_path, control_fn)\n",
    "my_map_file = os.path.join(local_path, layer_map)\n",
    "gds_in = os.path.join(local_path, gds_fn)"
   ]

    "only a part of the stackup is needed."

    "print(f\"EDB version: {version}\")\n",
    "\n",
    "edb = Edb(gds_in, edbversion=version, control_file=my_control_file, map_file=my_map_file)\n",
